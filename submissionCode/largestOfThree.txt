CLASS largestOfThree {
    VAR a, b, c;
    GET a; 
    GET b;
    GET c;

    IF a > b THEN {
        IF a > c THEN {
            PUT a;
        } 
        ELSE {
            PUT c;
        }
    }
    ELSE {
        IF b > c THEN {
            PUT b;
        }
        ELSE {
            PUT c;
        }
    }
}
